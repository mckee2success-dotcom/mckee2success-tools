// ─── Helpers ───────────────────────────────────────────────────────────────────
const callClaude = async (emailContent) => {
  const res = await fetch("https://api.anthropic.com/v1/messages", {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({
      model: "claude-sonnet-4-20250514",
      max_tokens: 1000,
      system: SYSTEM_PROMPT,
      messages: [{ role: "user", content: `Analyze this email and respond in JSON:\n\n${emailContent}` }],
    }),
  });
  const data = await res.json();
  const text = data.content?.map((b) => b.text || "").join("") || "{}";
  try {
    return JSON.parse(text.replace(/```json|```/g, "").trim());
  } catch {
    return { isJobRequisition: false, reason: "Parse error", reply: null };
  }
};

const callGmailSearch = async (query, maxResults = 15) => {
  const res = await fetch("https://api.anthropic.com/v1/messages", {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({
      model: "claude-sonnet-4-20250514",
      max_tokens: 1000,
      messages: [{ role: "user", content: query }],
      mcp_servers: [{ type: "url", url: "https://gmail.mcp.claude.com/mcp", name: "gmail-mcp" }],
    }),
  });
  const data = await res.json();
  return data;
};

const STATUS = { IDLE: "idle", SCANNING: "scanning", DONE: "done", ERROR: "error" };

// ─── Main Component ────────────────────────────────────────────────────────────
export default function GmailJobReply() {
  const [status, setStatus] = useState(STATUS.IDLE);
  const [logs, setLogs] = useState([]);
  const [emails, setEmails] = useState([]);
  const [selected, setSelected] = useState(null);
  const [drafting, setDrafting] = useState(null);
  const [draftResult, setDraftResult] = useState({});
  const [searchDays, setSearchDays] = useState(7);
  const [autoMode, setAutoMode] = useState(false);

  const log = useCallback((msg, type = "info") => {
    setLogs((prev) => [...prev, { msg, type, ts: new Date().toLocaleTimeString() }]);
  }, []);

  // ── Step 1: Search Gmail for job-related emails ──────────────────────────────
  const scanInbox = async () => {
    setStatus(STATUS.SCANNING);
    setLogs([]);
    setEmails([]);
    setSelected(null);
    setDraftResult({});

    log("🔍 Connecting to Gmail...");

    try {
      const searchPrompt = `Search my Gmail inbox for job requisitions, recruiter outreach, and job opportunities from the last ${searchDays} days. 
Use the Gmail search tool with these queries:
1. subject:(job OR position OR opportunity OR hiring OR recruiter OR role) is:unread newer_than:${searchDays}d
2. from:(recruit OR talent OR careers OR jobs OR hr OR hiring) newer_than:${searchDays}d
3. subject:(opening OR vacancy OR we're hiring OR join our team) newer_than:${searchDays}d

For each result, return the messageId, threadId, subject, from, and date. Return ONLY a JSON array of objects with fields: messageId, threadId, subject, from, date, snippet.`;

      const rawData = await callGmailSearch(searchPrompt);

      // Extract message list from MCP tool results
      const toolResults = rawData.content?.filter((b) => b.type === "mcp_tool_result") || [];
      const textBlocks = rawData.content?.filter((b) => b.type === "text") || [];

      log(`📬 Gmail responded. Parsing results...`);

      // Try to parse email list from text blocks
      let foundEmails = [];
      for (const block of [...toolResults, ...textBlocks]) {
        const text = block.content?.[0]?.text || block.text || "";
        // Look for JSON arrays
        const match = text.match(/\[[\s\S]*?\]/);
        if (match) {
          try {
            const parsed = JSON.parse(match[0]);
            if (Array.isArray(parsed) && parsed.length > 0 && parsed[0].messageId) {
              foundEmails = parsed;
              break;
            }
          } catch {}
        }
        // Look for message IDs inline
        const ids = text.match(/messageId[":]+\s*"?([a-zA-Z0-9]+)"?/g);
        if (ids && foundEmails.length === 0) {
          log(`📋 Found ${ids.length} potential message references`, "info");
        }
      }

      if (foundEmails.length === 0) {
        // Fall back: fetch full message list directly
        log("📋 Fetching message list directly...");
        const fetchRes = await callGmailSearch(
          `Use gmail_search_messages with query "subject:job OR subject:opportunity OR subject:hiring OR subject:position newer_than:${searchDays}d" and maxResults 20. Return the full JSON result.`
        );

        const allText = fetchRes.content?.map((b) => b.text || b.content?.[0]?.text || "").join("\n") || "";
        const matches = [...allText.matchAll(/"id"\s*:\s*"([a-zA-Z0-9]+)"/g)];
        const threadMatches = [...allText.matchAll(/"threadId"\s*:\s*"([a-zA-Z0-9]+)"/g)];
        const subjectMatches = [...allText.matchAll(/"subject"\s*:\s*"([^"]+)"/gi)];
        const fromMatches = [...allText.matchAll(/"from"\s*:\s*"([^"]+)"/gi)];
        const snippetMatches = [...allText.matchAll(/"snippet"\s*:\s*"([^"]+)"/gi)];

        for (let i = 0; i < Math.min(matches.length, 15); i++) {
          foundEmails.push({
            messageId: matches[i]?.[1],
            threadId: threadMatches[i]?.[1] || matches[i]?.[1],
            subject: subjectMatches[i]?.[1] || "(no subject)",
            from: fromMatches[i]?.[1] || "unknown",
            snippet: snippetMatches[i]?.[1] || "",
          });
        }
      }

      if (foundEmails.length === 0) {
        log("📭 No job emails found in the specified timeframe.", "warn");
        setStatus(STATUS.DONE);
        return;
      }

      log(`✅ Found ${foundEmails.length} potential job emails`, "success");

      // ── Step 2: Analyze each email with Claude ─────────────────────────────
      log("🤖 Analyzing emails with AI...");
      const analyzed = [];

      for (const email of foundEmails) {
        log(`  ↳ Analyzing: "${email.subject?.slice(0, 60)}..."`);
        const content = `Subject: ${email.subject}\nFrom: ${email.from}\nSnippet: ${email.snippet}`;
        const analysis = await callClaude(content);

        analyzed.push({
          ...email,
          ...analysis,
          id: email.messageId,
          status: "pending",
        });

        if (analysis.isJobRequisition) {
          log(`  ✓ JOB FOUND: "${email.subject?.slice(0, 50)}" @ ${analysis.company || "unknown"} [${analysis.confidence}]`, "success");
        }
      }

      const jobs = analyzed.filter((e) => e.isJobRequisition);
      const nonJobs = analyzed.filter((e) => !e.isJobRequisition);

      log(`\n📊 Summary: ${jobs.length} job requisitions found, ${nonJobs.length} skipped`);
      setEmails(analyzed);

      if (autoMode && jobs.length > 0) {
        log("\n🚀 Auto-mode: Creating drafts for all job emails...");
        for (const job of jobs) {
          await createDraft(job, analyzed, setEmails, setDraftResult, log);
        }
      }

      setStatus(STATUS.DONE);
    } catch (err) {
      log(`❌ Error: ${err.message}`, "error");
      setStatus(STATUS.ERROR);
    }
  };

  // ── Step 3: Create draft reply in Gmail ──────────────────────────────────────
  const createDraft = async (email, currentEmails, setEmailsFn, setDraftResultFn, logFn) => {
    if (!email.reply) return;
    setDrafting(email.id);
    logFn(`📝 Creating draft reply for "${email.subject?.slice(0, 50)}"...`);

    try {
      const draftRes = await callGmailSearch(
        `Create a Gmail draft reply to thread ${email.threadId} with this exact body text (plain text):

${email.reply}

Use the gmail_create_draft tool with:
- threadId: "${email.threadId}"
- body: the reply text above
- contentType: "text/plain"

Confirm when done.`
      );

      const allText = draftRes.content?.map((b) => b.text || b.content?.[0]?.text || "").join("\n") || "";
      const success = allText.toLowerCase().includes("draft") && (allText.toLowerCase().includes("created") || allText.toLowerCase().includes("success") || allText.toLowerCase().includes("saved"));

      if (success) {
        logFn(`✅ Draft created for "${email.subject?.slice(0, 40)}"`, "success");
        setEmailsFn((prev) =>
          prev.map((e) => (e.id === email.id ? { ...e, status: "drafted" } : e))
        );
        setDraftResultFn((prev) => ({ ...prev, [email.id]: "drafted" }));
      } else {
        logFn(`⚠️ Draft may not have saved — check Gmail Drafts manually`, "warn");
        setDraftResultFn((prev) => ({ ...prev, [email.id]: "uncertain" }));
      }
    } catch (err) {
      logFn(`❌ Draft error: ${err.message}`, "error");
      setDraftResultFn((prev) => ({ ...prev, [email.id]: "error" }));
    } finally {
      setDrafting(null);
    }
  };

  const jobEmails = emails.filter((e) => e.isJobRequisition);
  const otherEmails = emails.filter((e) => !e.isJobRequisition);

  // ── Render ────────────────────────────────────────────────────────────────────
  return (
    <div style={{
      minHeight: "100vh",
      background: "#0c0c10",
      color: "#e2e0db",
      fontFamily: "'Courier New', Courier, monospace",
      display: "flex",
      flexDirection: "column",
    }}>
      <style>{`
        @import url('https://fonts.googleapis.com/css2?family=Playfair+Display:wght@700;900&family=Inconsolata:wght@400;600&display=swap');
        * { box-sizing: border-box; margin: 0; padding: 0; }
        ::-webkit-scrollbar { width: 4px; }
        ::-webkit-scrollbar-track { background: #111; }
        ::-webkit-scrollbar-thumb { background: #333; border-radius: 2px; }
        @keyframes pulse { 0%,100%{opacity:1} 50%{opacity:0.4} }
        @keyframes scan { 0%{transform:translateY(-100%)} 100%{transform:translateY(400%)} }
        @keyframes fadeIn { from{opacity:0;transform:translateY(6px)} to{opacity:1;transform:translateY(0)} }
        @keyframes spin { to{transform:rotate(360deg)} }
        .email-row:hover { background: rgba(255,255,255,0.04) !important; }
        .action-btn:hover { opacity: 0.85; transform: translateY(-1px); }
        .day-btn:hover { background: rgba(255,255,255,0.1) !important; }
      `}</style>

      {/* ── Top Bar ── */}
      <div style={{
        borderBottom: "1px solid #1e1e26",
        padding: "18px 32px",
        display: "flex",
        alignItems: "center",
        justifyContent: "space-between",
        background: "rgba(0,0,0,0.4)",
      }}>
        <div>
          <div style={{ fontSize: "10px", letterSpacing: "3px", color: "#5a7a6a", marginBottom: "4px" }}>
            ENERO MCKEE · CISSP · PMP · CEH
          </div>
          <h1 style={{
            fontFamily: "'Playfair Display', serif",
            fontSize: "22px",
            fontWeight: "900",
            letterSpacing: "-0.5px",
            color: "#e8e4dc",
          }}>
            Job Reply Engine
          </h1>
        </div>
        <div style={{ display: "flex", alignItems: "center", gap: "8px" }}>
          <div style={{
            width: "8px", height: "8px", borderRadius: "50%",
            background: status === STATUS.SCANNING ? "#f59e0b" : status === STATUS.DONE ? "#34d399" : "#374151",
            animation: status === STATUS.SCANNING ? "pulse 1s infinite" : "none",
          }} />
          <span style={{ fontSize: "11px", color: "#4b5563", letterSpacing: "1px" }}>
            {status === STATUS.SCANNING ? "SCANNING" : status === STATUS.DONE ? "READY" : "STANDBY"}
          </span>
        </div>
      </div>

      <div style={{ display: "flex", flex: 1, overflow: "hidden", minHeight: 0 }}>

        {/* ── Left Panel: Controls + Log ── */}
        <div style={{
          width: "320px",
          flexShrink: 0,
          borderRight: "1px solid #1e1e26",
          display: "flex",
          flexDirection: "column",
          background: "#0a0a0e",
        }}>
          {/* Controls */}
          <div style={{ padding: "24px 20px", borderBottom: "1px solid #1a1a22" }}>
            <div style={{ fontSize: "9px", letterSpacing: "2px", color: "#4b5563", marginBottom: "16px" }}>
              SCAN SETTINGS
            </div>

            <div style={{ marginBottom: "16px" }}>
              <div style={{ fontSize: "11px", color: "#6b7280", marginBottom: "8px" }}>Search window</div>
              <div style={{ display: "flex", gap: "6px" }}>
                {[3, 7, 14, 30].map((d) => (
                  <button key={d} className="day-btn"
                    onClick={() => setSearchDays(d)}
                    style={{
                      flex: 1,
                      padding: "6px 0",
                      background: searchDays === d ? "rgba(52,211,153,0.15)" : "rgba(255,255,255,0.04)",
                      border: `1px solid ${searchDays === d ? "#34d399" : "#1e1e26"}`,
                      borderRadius: "4px",
                      color: searchDays === d ? "#34d399" : "#4b5563",
                      fontSize: "11px",
                      cursor: "pointer",
                      transition: "all 0.15s",
                    }}>
                    {d}d
                  </button>
                ))}
              </div>
            </div>

            <div style={{ marginBottom: "20px" }}>
              <label style={{ display: "flex", alignItems: "center", gap: "10px", cursor: "pointer" }}>
                <div
                  onClick={() => setAutoMode(!autoMode)}
                  style={{
                    width: "36px", height: "20px", borderRadius: "10px",
                    background: autoMode ? "#34d399" : "#1e1e26",
                    position: "relative", transition: "background 0.2s", cursor: "pointer",
                    border: "1px solid " + (autoMode ? "#34d399" : "#333"),
                  }}>
                  <div style={{
                    position: "absolute", top: "2px",
                    left: autoMode ? "17px" : "2px",
                    width: "14px", height: "14px", borderRadius: "50%",
                    background: "#fff", transition: "left 0.2s",
                  }} />
                </div>
                <span style={{ fontSize: "12px", color: "#6b7280" }}>
                  Auto-draft all jobs
                </span>
              </label>
            </div>

            <button
              onClick={scanInbox}
              disabled={status === STATUS.SCANNING}
              style={{
                width: "100%",
                padding: "12px",
                background: status === STATUS.SCANNING
                  ? "rgba(52,211,153,0.08)"
                  : "linear-gradient(135deg, #065f46, #047857)",
                border: "1px solid " + (status === STATUS.SCANNING ? "#065f46" : "#34d399"),
                borderRadius: "6px",
                color: status === STATUS.SCANNING ? "#34d399" : "#d1fae5",
                fontSize: "12px",
                fontFamily: "'Inconsolata', monospace",
                letterSpacing: "1px",
                cursor: status === STATUS.SCANNING ? "not-allowed" : "pointer",
                display: "flex", alignItems: "center", justifyContent: "center", gap: "8px",
                transition: "all 0.2s",
              }}>
              {status === STATUS.SCANNING ? (
                <>
                  <span style={{
                    width: "12px", height: "12px", border: "2px solid #34d39950",
                    borderTopColor: "#34d399", borderRadius: "50%",
                    animation: "spin 0.8s linear infinite", display: "inline-block",
                  }} />
                  SCANNING INBOX...

              ) : "▶ SCAN INBOX"}
            </button>

            </div>

                No activity yet. Click Scan Inbox to begin.

            {logs.map((l, i) => (

                { label: "Scanned", val: emails.length, color: "#6b7280" },
                { label: "Jobs", val: jobEmails.length, color: "#34d399" },
                { label: "Drafted", val: Object.values(draftResult).filter(v => v === "drafted").length, color: "#60a5fa" },
              ].map(({ label, val, color }) => (
                <div key={label} style={{ textAlign: "center", flex: 1 }}>
                  <div style={{ fontSize: "20px", fontWeight: "700", color, fontFamily: "'Playfair Display', serif" }}>{val}</div>
                  <div style={{ fontSize: "9px", letterSpacing: "1px", color: "#374151" }}>{label.toUpperCase()}</div>

          {jobEmails.map((email) => (

          {otherEmails.length > 0 && (
