const GMAIL_MCP = "https://gmailmcp.googleapis.com/mcp/v1";
const MODEL = "claude-sonnet-4-20250514";

// ── Helpers ─────────────────────────────────────────────────────────
async function callClaude(messages, systemPrompt, useMCP = true) {
  const body = {
    model: MODEL,
    max_tokens: 1000,
    system: systemPrompt,
    messages,
  };
  if (useMCP) body.mcp_servers = [{ type: "url", url: GMAIL_MCP, name: "gmail-mcp" }];

  const res = await fetch("https://api.anthropic.com/v1/messages", {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify(body),
  });
  return res.json();
}

function extractText(data) {
  return (data.content || [])
    .filter((b) => b.type === "text")
    .map((b) => b.text)
    .join("\n");
}

function extractToolResults(data) {
  return (data.content || [])
    .filter((b) => b.type === "mcp_tool_result")
    .map((b) => b.content?.[0]?.text || "")
    .join("\n");
}
