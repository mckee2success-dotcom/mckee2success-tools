    const grade = overallPct >= 80 ? { label: "PASSING", color: "#22c55e" } : overallPct >= 65 ? { label: "BORDERLINE", color: "#f59e0b" } : { label: "NEEDS WORK", color: "#ef4444" };
    return (
      <div style={S.root}>
        <style>@import url('https://fonts.googleapis.com/css2?family=JetBrains+Mono:wght@400;500;600;700&display=swap');</style>
        <div style={S.topBar}>
          <div style={S.brand}><div style={S.brandIcon}>⬡</div><div><div style={S.brandText}>CMMC CPC Exam</div><div style={S.brandSub}>Certified CMMC Professional</div></div></div>
          <div style={S.tabs}>{["test","map","results"].map(t => <button key={t} style={S.tab(tab===t)} onClick={() => setTab(t)}>{t}</button>)}</div>
        </div>
        <div style={{ maxWidth: "900px", margin: "30px auto", padding: "0 20px" }}>
          <div style={{ display: "grid", gridTemplateColumns: "repeat(4,1fr)", gap: "12px", marginBottom: "24px" }}>
            {[
              { val: totalAnswered, lbl: "Attempted", color: "#5080a0" },
              { val: totalCorrect, lbl: "Correct", color: "#22c55e" },
              { val: totalAnswered - totalCorrect, lbl: "Incorrect", color: "#ef4444" },
              { val: `${overallPct}%`, lbl: grade.label, color: grade.color },
            ].map((s,i) => (
              <div key={i} style={S.resultStat}>
                <div style={S.resultStatVal(s.color)}>{s.val}</div>
                <div style={S.resultStatLbl}>{s.lbl}</div>
              </div>
            ))}
          </div>
          <div style={{ background: "#0d1220", border: "1px solid rgba(255,255,255,0.05)", borderRadius: "10px", padding: "20px" }}>
            <div style={{ ...S.sideLabel, marginBottom: "14px" }}>Performance by Domain</div>
            {Object.entries(byDomain).map(([domain, { total, correct }]) => {
              const h = DOMAIN_HUE[domain] || 200;
              const p = Math.round((correct / total) * 100);
              const answered = Object.keys(answers).filter(id => ALL_QUESTIONS.find(q => q.id === parseInt(id))?.domain === domain).length;
              return (
                <div key={domain} style={{ marginBottom: "12px" }}>
                  <div style={{ display: "flex", justifyContent: "space-between", marginBottom: "4px" }}>
                    <span style={{ fontSize: "11px", color: "#7090b0" }}>{domain}</span>
                    <span style={{ fontSize: "11px", color: answered === 0 ? "#2a4060" : p >= 70 ? "#22c55e" : "#ef4444" }}>
                      {answered === 0 ? "—" : `${correct}/${answered}`}
                    </span>
                  </div>
                  <div style={{ height: "4px", background: "rgba(255,255,255,0.05)", borderRadius: "2px" }}>
                    {answered > 0 && <div style={{ height: "100%", width: `${p}%`, background: p >= 70 ? "#22c55e" : p >= 50 ? "#f59e0b" : "#ef4444", borderRadius: "2px", transition: "width 0.5s" }} />}
                  </div>
                </div>
              );
            })}
          </div>
          <div style={{ textAlign: "center", marginTop: "20px" }}>
            <button style={{ ...S.btnSubmit(false), padding: "11px 30px" }} onClick={() => { setTab("test"); goTo(0); }}>← Back to Test</button>
          </div>
        </div>
      </div>
    );
  }

  // ── MAP TAB ──
  if (tab === "map") {
    return (
      <div style={S.root}>
        <style>@import url('https://fonts.googleapis.com/css2?family=JetBrains+Mono:wght@400;500;600;700&display=swap');</style>
        <div style={S.topBar}>
          <div style={S.brand}><div style={S.brandIcon}>⬡</div><div><div style={S.brandText}>CMMC CPC Exam</div><div style={S.brandSub}>Certified CMMC Professional</div></div></div>
          <div style={S.tabs}>{["test","map","results"].map(t => <button key={t} style={S.tab(tab===t)} onClick={() => setTab(t)}>{t}</button>)}</div>
        </div>
        <div style={{ maxWidth: "900px", margin: "24px auto", padding: "0 20px" }}>
          <div style={{ background: "#0d1220", border: "1px solid rgba(255,255,255,0.05)", borderRadius: "10px", padding: "20px", marginBottom: "16px" }}>
            <div style={S.sideLabel}>All 25 Questions</div>
            <div style={{ display: "grid", gridTemplateColumns: "repeat(auto-fill, minmax(140px, 1fr))", gap: "8px" }}>
              {ALL_QUESTIONS.map((question, i) => {
                const ans = answers[question.id];
                const isCorrect = ans !== undefined && question.correct === ans;
                const isFlagged = flagged.has(question.id);
                const dh = DOMAIN_HUE[question.domain] || 200;
                return (
                  <div key={question.id}
                    onClick={() => { setTab("test"); setDomainFilter("All Domains"); const idx = filtered.findIndex(q => q.id === question.id); if (idx >= 0) goTo(idx); else { setDomainFilter("All Domains"); setTimeout(() => { const idx2 = ALL_QUESTIONS.findIndex(q => q.id === question.id); goTo(idx2); setTab("test"); }, 50); } }}
                    style={{
                      padding: "10px",
                      borderRadius: "8px",
                      border: ans !== undefined ? `1px solid ${isCorrect ? "rgba(34,197,94,0.3)" : "rgba(239,68,68,0.3)"}` : isFlagged ? "1px solid rgba(245,158,11,0.25)" : "1px solid rgba(255,255,255,0.05)",
                      background: ans !== undefined ? (isCorrect ? "rgba(34,197,94,0.06)" : "rgba(239,68,68,0.06)") : "rgba(255,255,255,0.02)",
                      cursor: "pointer",
                    }}>
                    <div style={{ fontSize: "9px", color: "#2a4060", marginBottom: "3px" }}>Q{question.id} · {question.difficulty}</div>
                    <div style={{ fontSize: "10px", color: ans !== undefined ? (isCorrect ? "#22c55e" : "#ef4444") : isFlagged ? "#f59e0b" : hsl(dh,50,60), letterSpacing: "0.05em", overflow: "hidden", textOverflow: "ellipsis", whiteSpace: "nowrap" }}>{question.domain}</div>
                  </div>
                );
              })}
            </div>
          </div>
          <div style={{ display: "flex", gap: "16px", justifyContent: "center", fontSize: "10px", color: "#2a4060" }}>
            {[["rgba(34,197,94,0.3)","Correct"],["rgba(239,68,68,0.3)","Incorrect"],["rgba(245,158,11,0.25)","Flagged"],["rgba(255,255,255,0.05)","Not Attempted"]].map(([c,l]) => (
              <span key={l} style={{ display: "flex", alignItems: "center", gap: "5px" }}><span style={{ width: "10px", height: "10px", borderRadius: "2px", border: `1px solid ${c}`, display: "inline-block" }} />{l}</span>
            ))}
          </div>
        </div>
      </div>
    );
  }

  // ── TEST TAB ──
  return (
    <div style={S.root} ref={topRef}>
      <style>@import url('https://fonts.googleapis.com/css2?family=JetBrains+Mono:wght@400;500;600;700&display=swap');</style>
      <div style={S.topBar}>
        <div style={S.brand}>
          <div style={S.brandIcon}>⬡</div>
          <div>
            <div style={S.brandText}>CMMC CPC Exam</div>
            <div style={S.brandSub}>Certified CMMC Professional · v2.0</div>
          </div>
        </div>
        <div style={S.tabs}>{["test","map","results"].map(t => <button key={t} style={S.tab(tab===t)} onClick={() => setTab(t)}>{t}</button>)}</div>
        <div style={S.scoreBar}>
          {[
            { val: `${totalCorrect}/${totalAnswered}`, lbl: "Score", c: pct === null ? "#2a4060" : pct >= 70 ? "#22c55e" : "#ef4444" },
            { val: pct === null ? "—" : `${pct}%`, lbl: "Pass Rate", c: pct === null ? "#2a4060" : pct >= 70 ? "#22c55e" : pct >= 50 ? "#f59e0b" : "#ef4444" },
            { val: flagged.size, lbl: "Flagged", c: flagged.size > 0 ? "#f59e0b" : "#2a4060" },
          ].map((s, i) => (
            <div key={i} style={{ display: "flex", alignItems: "center", gap: "16px" }}>
              {i > 0 && <div style={S.divider} />}
              <div style={S.scoreItem}>
                <div style={S.scoreVal(s.c)}>{s.val}</div>
                <div style={S.scoreLbl}>{s.lbl}</div>
              </div>
            </div>
          ))}
        </div>
      </div>

      <div style={S.body}>
        {/* Sidebar */}
        <div style={S.sidebar}>
          <div style={S.sideSection}>
            <div style={S.sideLabel}>Domain Filter</div>
            {DOMAINS.map(d => (
              <button key={d} style={S.filterBtn(domainFilter === d, d === "All Domains" ? 200 : DOMAIN_HUE[d] || 200)} onClick={() => setDomainFilter(d)}>
                {d}
              </button>
            ))}
          </div>
          <div style={S.sideSection}>
            <div style={S.sideLabel}>Difficulty</div>
            {DIFFICULTIES.map(d => {
              const dh = { Easy: 140, Medium: 45, Hard: 0, "All Levels": 200 }[d] || 200;
              return <button key={d} style={S.filterBtn(diffFilter === d, dh)} onClick={() => setDiffFilter(d)}>{d}</button>;
            })}
          </div>
          <div style={S.sideSection}>
            <div style={S.sideLabel}>Question Navigator</div>
            <div style={{ display: "grid", gridTemplateColumns: "repeat(5,1fr)", gap: "4px" }}>
              {filtered.map((fq, i) => (
                <button key={fq.id} style={S.mapBtn(fq.id, q.id)} onClick={() => goTo(i)}>
                  {fq.id}
                </button>
              ))}
            </div>
            <div style={{ marginTop: "12px", display: "flex", flexDirection: "column", gap: "4px", fontSize: "9px" }}>
              {[["#22c55e","Correct"],["#ef4444","Wrong"],["#f59e0b","Flagged"]].map(([c,l]) => (
                <span key={l} style={{ color: "#2a4060", display: "flex", alignItems: "center", gap: "5px" }}><span style={{ color: c }}>■</span>{l}</span>
              ))}
            </div>
          </div>
        </div>

        {/* Main */}
        <div>
          {/* Progress bar */}
          <div style={{ height: "3px", background: "rgba(255,255,255,0.04)", borderRadius: "2px", marginBottom: "14px", overflow: "hidden" }}>
            <div style={{ height: "100%", width: `${((qIndex + 1) / filtered.length) * 100}%`, background: `linear-gradient(90deg, ${hsl(hue,65,50)}, ${hsl(hue+20,60,45)})`, borderRadius: "2px", transition: "width 0.3s, background 0.4s" }} />
          </div>

          <div style={S.mainCard}>
            {/* Card header */}
            <div style={S.cardHeader}>
              <div>
                <div style={S.qCounter}>QUESTION {qIndex + 1} OF {filtered.length} · ID #{q.id}</div>
                <div style={S.metaRow}>
                  <span style={S.chip(hue)}>{q.domain}</span>
                  <span style={S.diffChip(q.difficulty)}>{q.difficulty}</span>
                  <span style={{ ...S.chip(0,0,50), background: "rgba(255,255,255,0.04)", color: "#3a5070" }}>CMMC {q.level}</span>
                  {answers[q.id] !== undefined && (
                    <span style={S.chip(answers[q.id] === q.correct ? 140 : 0, 65)}>
                      {answers[q.id] === q.correct ? "✓ CORRECT" : "✗ INCORRECT"}
                    </span>
                  )}
                </div>
              </div>
              <button style={S.flagBtn(flagged.has(q.id))} onClick={toggleFlag}>
                {flagged.has(q.id) ? "★" : "☆"}
              </button>
            </div>

            {/* Question */}
            <div style={S.qText}>{q.question}</div>

            {/* Options */}
            <div style={S.optWrap}>
              {q.options.map((opt, idx) => (
                <div key={idx} style={S.opt(idx, selected, revealed, q.correct)} onClick={() => handleSelect(idx)}>
                  <div style={S.badge(idx, selected, revealed, q.correct)}>
                    {revealed && idx === q.correct ? "✓" : revealed && idx === selected && idx !== q.correct ? "✗" : String.fromCharCode(65 + idx)}
                  </div>
                  <span>{opt}</span>
                </div>
              ))}
            </div>

            {/* Explanation */}
            {revealed && (
              <div style={S.expBox}>
                <div style={S.expTitle}>
                  <span>▸</span> Technical Explanation
                </div>
                <div style={S.expText}>{q.explanation}</div>
                <div style={S.refBadge}>
                  <span>📎</span>
                  <span>{q.ref}</span>
                </div>
              </div>
            )}

            {/* Footer */}
            <div style={S.footer}>
              <div style={S.navGroup}>
                <button style={S.btnNav(qIndex === 0)} onClick={handlePrev} disabled={qIndex === 0}>← Prev</button>
                <button style={S.btnNav(qIndex === filtered.length - 1)} onClick={handleNext} disabled={qIndex === filtered.length - 1}>Next →</button>
              </div>
              <div style={{ display: "flex", gap: "8px", alignItems: "center" }}>
                <span style={{ fontSize: "10px", color: "#1a3050" }}>{qIndex + 1} / {filtered.length}</span>
                {!revealed
                  ? <button style={S.btnSubmit(selected === null)} onClick={handleSubmit} disabled={selected === null}>Submit Answer</button>
                  : <button style={S.btnSubmit(qIndex === filtered.length - 1)} onClick={handleNext} disabled={qIndex === filtered.length - 1}>Next →</button>
                }
              </div>
            </div>
          </div>

          {/* Running score */}
          {totalAnswered > 0 && (
            <div style={{ marginTop: "12px", background: "#0d1220", border: "1px solid rgba(255,255,255,0.05)", borderRadius: "10px", padding: "14px 18px", display: "flex", alignItems: "center", gap: "16px" }}>
              <div>
                <div style={{ fontSize: "9px", color: "#2a4060", letterSpacing: "0.15em", textTransform: "uppercase", marginBottom: "2px" }}>Running Score</div>
                <div style={{ fontSize: "22px", fontWeight: "700", color: pct >= 70 ? "#22c55e" : pct >= 50 ? "#f59e0b" : "#ef4444" }}>{pct}%</div>
              </div>
              <div style={{ flex: 1, height: "6px", background: "rgba(255,255,255,0.04)", borderRadius: "3px", overflow: "hidden" }}>
                <div style={{ height: "100%", width: `${pct}%`, background: pct >= 70 ? "linear-gradient(90deg,#22c55e,#16a34a)" : pct >= 50 ? "linear-gradient(90deg,#f59e0b,#d97706)" : "linear-gradient(90deg,#ef4444,#dc2626)", borderRadius: "3px", transition: "width 0.4s" }} />
              </div>
              <div style={{ fontSize: "11px", color: "#3a5070", textAlign: "right" }}>
                <div>{totalCorrect} correct</div>
                <div style={{ color: "#1a3050" }}>{totalAnswered - totalCorrect} wrong</div>
              </div>
            </div>
          )}
        </div>
      </div>
    </div>
  );
}
